### 建议 1：增加"为什么"的设计意图说明

#### 当前问题

#### 改进方案

// ✅ 说明"为什么"
// ════ 单实例保护机制 ════
// 原因：多个插件实例同时运行会导致：
// 1. 多个线程竞争同一任务栏窗口，引起画面闪烁
// 2. WebSocket 连接冲突，消息丢失
// 3. 资源重复占用，导致内存泄漏
// 
// 实现：使用全局 Mutex，EXE 启动时尝试创建同名 Mutex
// 如果已存在则说明前一实例仍在运行，直接退出
::CreateMutexW(nullptr, FALSE, L"MoeKoeTaskbarLyrics_Mutex");
if (::GetLastError() == ERROR_ALREADY_EXISTS) {
    DebugLog("[STARTUP] Another instance already running, exiting gracefully\n");
    return 0;
}

#### 改进方案

创建 `src/constants.h`：

```cpp
#pragma once

namespace moekoe::config {

// ════ 网络通信端口 ════
/// MoeKoeMusic WebSocket 服务端口（用于接收歌词数据）
/// 参考：MoeKoeMusic 源代码中的 WebSocket 服务器初始化
constexpr int WEBSOCKET_LISTEN_PORT = 6520;

/// 本插件 HTTP 服务器端口（用于 Chrome Extension popup.js 通信）
/// ⚠️ 修改此端口时需同步修改 moeKoe-taskbar-lyrics/popup.js
constexpr int HTTP_SERVER_PORT = 6523;

// ════ 渲染相关常量 ════
/// 帧定时器最小间隔（毫秒）
/// 限制原因：过快会导致过度渲染，浪费 CPU；过慢会掉帧
/// 计算方式：MAX(MIN_FRAME_INTERVAL, 1000 / 刷新率Hz)
constexpr int MIN_FRAME_INTERVAL_MS = 15;

/// 目标帧率默认值（刷新率，单位 Hz）
/// 推荐值：30 FPS（平衡 CPU 占用和渲染流畅度）
constexpr int DEFAULT_REFRESH_RATE_HZ = 30;

// ════ UI 相关常量 ════
/// Windows Tooltip 最大字符数
/// 参考：Microsoft 官方文档 WM_SETTEXT
/// 注意：包含 null terminator，超过此数会被系统截断
constexpr int WINDOWS_TOOLTIP_MAX_LEN = 127;

// ════ 进程监控相关常量 ════
/// 进程监控轮询间隔（毫秒）
/// 用于绑定模式检测 MoeKoeMusic.exe 的启动/退出
constexpr int PROCESS_MONITOR_INTERVAL_MS = 2000;

// ════ 调试相关常量 ════
/// 日志文件名
constexpr const char* DEBUG_LOG_FILENAME = "debug.log";

} // namespace moekoe::config
```

然后在 main.cpp 中使用：

```cpp
#include "constants.h"
using namespace moekoe::config;

// ...

if (httpServer.Start(HTTP_SERVER_PORT)) {
    DebugLog("[STARTUP] HTTP server started on port %d\n", HTTP_SERVER_PORT);
}

// 更新帧定时器

    MIN_FRAME_INTERVAL_MS, 
    1000 / std::max(1, config.Advanced().refreshRateHz)

::SetTimer(hMsgWnd, 1, static_cast<UINT>(intervalMs), nullptr);

    WSADATA wsaData;
    int wsRet = WSAStartup(MAKEWORD(2, 2), &wsaData);
    DebugLog("[STARTUP] WSAStartup ret=%d\n", wsRet);
    
    ::SetProcessDpiAwarenessContext(DPI_AWARENESS_CONTEXT_PER_MONITOR_AWARE_V2);
    
    moekoe::Config config;
    config.Load();
    DebugLog("[STARTUP] Config loaded\n");
    config.SetAutoStart(config.IsAutoStart());
    
    // ... 消息窗口创建、托盘初始化等
    
    // ════ 第 3 阶段：业务模块初始化 ════
    // 目的：创建核心业务逻辑模块
    // 包括：任务栏窗口、WebSocket 客户端、渲染引擎等
    HWND hTaskbar = moekoe::TaskbarWindow::FindTaskbarHandle();
    if (!hTaskbar) {
        ::MessageBoxW(nullptr, L"未找到 Windows 任务栏...", L"...", MB_OK | MB_ICONERROR);
        return 1;
    }
    
    moekoe::TaskbarWindow taskbarWindow;
    if (!taskbarWindow.Create(hInstance, hTaskbar)) {
        return 1;
    }
    
    moekoe::TaskbarRenderer renderer;
    if (!renderer.Initialize(taskbarWindow.GetHandle())) {
        return 1;
    }
    
    // ... WebSocket 连接、HTTP 服务器启动等
    
    // ════ 第 4 阶段：业务逻辑循环 ════
    // 目的：处理消息和事件，直到应用关闭
    MSG msg{};
    while (app.running && ::GetMessageW(&msg, nullptr, 0, 0)) {
        ::TranslateMessage(&msg);
        ::DispatchMessageW(&msg);
    }
    DebugLog("[SHUTDOWN] Message loop ended\n");
    
    // ════ 第 5 阶段：清理和退出 ════
    // 目的：释放所有资源，进行优雅关闭
    // 注意：顺序很重要，应与初始化顺序相反
    ::KillTimer(hMsgWnd, 1);
    httpServer.Stop();              // 停止 HTTP 服务
    processMonitor.Stop();          // 停止进程监控
    wsClient.Disconnect();          // 断开 WebSocket
    renderer.Shutdown();            // 释放 Direct2D 资源
    taskbarWindow.Destroy();        // 销毁窗口
    tray.Shutdown();                // 销毁托盘图标
    ::DestroyWindow(hMsgWnd);       // 销毁消息窗口
    
    DebugLog("[SHUTDOWN] Complete\n");
    return 0;
}
```

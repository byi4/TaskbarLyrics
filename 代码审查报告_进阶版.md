```cpp
void WebSocketClient::ReconnectLoop() {
    int attempt = 0;
    while (!stopRequested_.load()) {
        // 已连接：持续监控
        if (connected_.load() && client_) {
            std::this_thread::sleep_for(
                std::chrono::milliseconds(constants::WS_CONNECTED_POLL_MS));
            continue;
        }

        // 指数退避
        const int waitSec = BackoffSeconds(attempt++);
        // ...
        
        // 5 秒连接超时
        for (int i = 0; i < constants::WS_CONNECT_TIMEOUT_ITERATIONS && !stopRequested_.load(); ++i) {
            if (connected_.load()) break;
            std::this_thread::sleep_for(
                std::chrono::milliseconds(constants::RECONNECT_WAIT_GRANULARITY_MS));
        }
    }
}

int BackoffSeconds(int attempt) {
    if (attempt <= 0) return 1;
    if (attempt == 1) return 1;
    if (attempt == 2) return 2;
    if (attempt == 3) return 4;
    if (attempt == 4) return 8;
    return 15;  // 上限 15 秒
}
```

**优点：**
- 指数退避避免过度重连
- 5 秒连接窗口防止无限等待
- 可中断（支持 stopRequested_）

void WebSocketClient::Disconnect() {
    stopRequested_.store(true);
    reconnectNow_.store(false);

    if (client_) {
        try {
            client_->stop();
        } catch (...) { /* ignore */ }

    if (reconnectThread_.joinable()) {

int LyricsParser::FindLineIndex(double currentTimeSec) const {
    const int64_t tMs = static_cast<int64_t>(currentTimeSec * 1000.0);
    const int n = static_cast<int>(lyrics_.lines.size());
    int lo = 0, hi = n - 1, best = -1;
    
    while (lo <= hi) {
        const int mid = (lo + hi) / 2;
        const auto& chars = lyrics_.lines[mid].characters;
        if (chars.empty()) {
            lo = mid + 1;
            continue;
        }
        const int64_t startMs = chars.front().startTime;
        if (startMs <= tMs) {
            best = mid;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }
    return best;
}

#### 5.2.2 精确的字符级进度计算（第 81-117 行）

```cpp
if (!line.characters.empty()) {
    const int64_t tMs = static_cast<int64_t>(state_.currentTime * 1000.0);
    const auto& chars = line.characters;
    
    // 找到 tMs 落入哪个字符
    int charIdx = -1;
    for (size_t i = 0; i < chars.size(); ++i) {
        if (tMs >= chars[i].startTime && tMs <= chars[i].endTime) {
            charIdx = static_cast<int>(i);
            break;
        }
        // 处理时间缝隙
        if (tMs > chars[i].endTime &&
            (i + 1 >= chars.size() || tMs < chars[i + 1].startTime)) {
            charIdx = static_cast<int>(i);
            break;
        }
    }
    
    // 计算进度
    if (charIdx < 0) {
        out.progress = 0.0;
    } else if (static_cast<size_t>(charIdx) >= chars.size() - 1) {
        out.progress = 1.0;
    } else {
        // 已唱字符数 + 当前字符内进度
        const auto& cur = chars[charIdx];
        const int64_t dur = std::max<int64_t>(1, cur.endTime - cur.startTime);
        const double inside = std::clamp(
            static_cast<double>(tMs - cur.startTime) / static_cast<double>(dur),
            0.0, 1.0);
        out.progress = (static_cast<double>(charIdx) + inside) /
                       static_cast<double>(chars.size());
    }
}
```

**优点：**
- 处理时间缝隙（字符间的间隙）
- 计算字符内部进度
- 避免浮点数除以零

std::string GetDebugLogPath() {
    static std::string path;
